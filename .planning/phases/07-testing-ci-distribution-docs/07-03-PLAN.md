Create .github/workflows/ci.yml with two jobs:

**Job 1: lint-and-unit (runs on every PR and push to main) — CICD-01**
```yaml
name: CI
on:
  pull_request:
    branches: [main]
  push:
    branches: [main]

jobs:
  lint-and-unit:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Install dependencies
        run: |
          sudo apt-get update
          sudo apt-get install -y shellcheck
          # Install bats
          git clone --depth 1 https://github.com/bats-core/bats-core.git /tmp/bats
          sudo /tmp/bats/install.sh /usr/local
          # Install bats helpers
          git clone --depth 1 https://github.com/bats-core/bats-support.git test/helpers/bats-support
          git clone --depth 1 https://github.com/bats-core/bats-assert.git test/helpers/bats-assert
          # Install yq (needed by scenario.sh)
          sudo wget -qO /usr/local/bin/yq https://github.com/mikefarah/yq/releases/download/v4.44.6/yq_linux_amd64
          sudo chmod +x /usr/local/bin/yq
      - name: Shellcheck
        run: make shellcheck
      - name: Unit tests
        run: make test-unit
```

**Job 2: integration (runs only on push to main, NOT on PRs) — CICD-02**
```yaml
  integration:
    runs-on: ubuntu-latest
    if: github.event_name == 'push' && github.ref == 'refs/heads/main'
    needs: lint-and-unit
    steps:
      - uses: actions/checkout@v4
      - name: Install dependencies
        run: |
          sudo apt-get update
          sudo apt-get install -y shellcheck
          git clone --depth 1 https://github.com/bats-core/bats-core.git /tmp/bats
          sudo /tmp/bats/install.sh /usr/local
          git clone --depth 1 https://github.com/bats-core/bats-support.git test/helpers/bats-support
          git clone --depth 1 https://github.com/bats-core/bats-assert.git test/helpers/bats-assert
          sudo wget -qO /usr/local/bin/yq https://github.com/mikefarah/yq/releases/download/v4.44.6/yq_linux_amd64
          sudo chmod +x /usr/local/bin/yq
          # Install kind
          curl -Lo ./kind https://kind.sigs.k8s.io/dl/v0.25.0/kind-linux-amd64
          chmod +x ./kind
          sudo mv ./kind /usr/local/bin/kind
          # Install jq (usually pre-installed on ubuntu-latest but ensure)
          sudo apt-get install -y jq
      - name: Create kind cluster
        run: |
          kind create cluster --name ckad-drill --wait 60s
      - name: Shellcheck
        run: make shellcheck
      - name: Unit tests
        run: make test-unit
      - name: Integration tests
        run: make test-integration
      - name: Cleanup
        if: always()
        run: kind delete cluster --name ckad-drill 2>/dev/null || true
```

Ensure the `.github/workflows/` directory is created. The integration job uses `needs: lint-and-unit` so it only runs if lint+unit passed. The `if` condition ensures integration only runs on merge to main.

Note: The CI cluster does NOT install Calico/ingress/metrics-server (full `ckad-drill start` cluster) — it uses a minimal `kind create cluster`. Integration tests should handle this by skipping tests that require those addons, or using the simpler scenarios that don't need network policies or ingress.
