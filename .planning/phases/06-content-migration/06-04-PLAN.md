Example learn scenario:
```yaml
id: learn-pods-volumes
domain: 1
title: "Pods, Containers, and Shared Volumes"
difficulty: easy
time_limit: 180
namespace: learn-pods
learn_intro: >
  A pod is a group of one or more containers that share network (same IP,
  localhost between containers) and storage (volumes declared at pod level,
  mounted by each container). emptyDir volumes exist for the pod's lifetime.
  Generate pod YAML fast: kubectl run mypod --image=nginx --dry-run=client -o yaml > pod.yaml
description: >
  Create a pod named "web" using nginx. Add an emptyDir volume named "data"
  mounted at /usr/share/nginx/html. Verify the pod is running.
hint: >
  kubectl run web --image=nginx --dry-run=client -o yaml > pod.yaml, then add
  the volume and volumeMount sections.
validations:
  - name: pod_exists
    type: resource_exists
    resource: pod/web
    namespace: learn-pods
  - name: volume_mounted
    type: volume_mount
    pod: web
    container: web
    namespace: learn-pods
    mount_path: /usr/share/nginx/html
solution:
  steps:
    - "kubectl create ns learn-pods --dry-run=client -o yaml | kubectl apply -f -"
    - "kubectl run web --image=nginx --dry-run=client -o yaml > /tmp/pod.yaml"
    - "# Edit pod.yaml to add emptyDir volume and volumeMount"
    - "kubectl apply -f /tmp/pod.yaml -n learn-pods"
```
