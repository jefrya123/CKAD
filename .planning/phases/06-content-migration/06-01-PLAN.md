<!-- YAML scenario schema consumed by lib/scenario.sh -->
Required top-level fields:
  id: string (unique, prefix sc-)
  domain: integer (1-5)
  title: string
  difficulty: easy|medium|hard
  time_limit: integer (seconds)
  namespace: string (realistic name, NOT drill-<id>)
  description: string (multi-line block scalar)
  hint: string (multi-line block scalar)
  validations: array of typed checks
  solution: object with steps array

Validation check types available (from lib/validator.sh):
  - resource_exists: { resource, namespace }
  - resource_field: { resource, namespace, jsonpath, expected }
  - container_count: { pod, namespace, expected }
  - container_image: { pod, container, namespace, expected }
  - container_env: { pod, container, namespace, env_var, expected }
  - volume_mount: { pod, container, namespace, mount_path }
  - container_running: { pod, namespace }
  - label_selector: { resource_type, namespace, selector, [min_count] }
  - resource_count: { resource_type, namespace, selector, expected }
  - command_output: { command, mode (contains|matches|equals), expected }

Optional fields:
  tags: [helm] — triggers Helm check (SCEN-06)
  learn_intro: string — enables learn mode (Phase 5)
  setup: { commands: [], manifest: string } — pre-setup resources
</interfaces>
