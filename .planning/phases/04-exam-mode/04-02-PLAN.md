  <action>
  Add `progress_record_exam` function to `lib/progress.sh`:

  ```bash
  # progress_record_exam SCORE PASSED DOMAIN_RESULTS_JSON
  # Appends an exam result to the .exams array in progress.json.
  # SCORE: integer percentage (0-100)
  # PASSED: "true" or "false"
  # DOMAIN_RESULTS_JSON: JSON string of per-domain results from exam_grade
  progress_record_exam() {
    local score="$1"
    local passed="$2"
    local domain_results="$3"

    progress_init

    local timestamp
    timestamp=$(date -u +"%Y-%m-%dT%H:%M:%SZ")

    local tmp_file
    tmp_file="${CKAD_PROGRESS_FILE}.tmp.$$"

    jq --argjson score "${score}" \
       --argjson passed "${passed}" \
       --arg ts "${timestamp}" \
       --argjson domains "${domain_results}" \
      '.exams += [{
        "date": $ts,
        "score": $score,
        "passed": $passed,
        "domains": $domains
      }]' \
      "${CKAD_PROGRESS_FILE}" > "${tmp_file}" && mv "${tmp_file}" "${CKAD_PROGRESS_FILE}"
  }
  ```

  Add test to `test/unit/progress.bats`:
  - Test that progress_record_exam appends to .exams array
  - Test that multiple calls accumulate entries
  - Verify JSON structure has date, score, passed, domains fields

  **Step 3: Add exam subcommand to case dispatch**
  Add before the `*)` catch-all:

  ```bash
  exam)
    shift
    case "${1:-}" in
      # ckad-drill exam (no sub-subcommand) — start a new exam
      ""|start)
        shift 2>/dev/null || true
        # Parse flags
        local exam_time=7200  # default 2 hours
        local exam_count=16   # default 16 questions
        while [[ $# -gt 0 ]]; do
          case "$1" in
            --time) exam_time="$2"; shift 2 ;;
            --count) exam_count="$2"; shift 2 ;;
            *) error "Unknown flag: $1"; exit "${EXIT_ERROR}" ;;
          esac
        done

        cluster_check_active || exit "${EXIT_NO_CLUSTER}"

        # Select questions with CKAD domain weighting
        local -a question_files=()
        mapfile -t question_files < <(exam_select_questions "${exam_count}")

        if [[ "${#question_files[@]}" -eq 0 ]]; then
          error "No scenarios available for exam."
          exit "${EXIT_ERROR}"
        fi

        info "Starting exam with ${#question_files[@]} questions (${exam_time}s time limit)..."

        # Install exam cleanup trap
        trap '_exam_cleanup' INT TERM EXIT

        # Write exam session
        local exam_end_at
        exam_end_at=$(( $(date +%s) + exam_time ))
        exam_session_write "${exam_end_at}" "${exam_time}" "${question_files[@]}"

        # Create all namespaces upfront (EXAM-02)
        exam_setup_all_namespaces

        # Display first question
        exam_session_read
        scenario_load "${EXAM_CURRENT_FILE}"
        _drill_display "${EXAM_CURRENT_FILE}"

        info ""
        info "Exam started! ${#question_files[@]} questions, $(( exam_time / 60 )) minutes."
        info "Commands: exam list | exam next | exam prev | exam jump N | exam flag | check | exam submit"
        info "Run: source <(ckad-drill env) to start timer"

        # Remove EXIT trap — session established
        trap - EXIT
        ;;

      list)   # EXAM-04
        exam_list
        ;;

      next)   # EXAM-05
        exam_navigate "next"
        exam_session_read
        scenario_load "${EXAM_CURRENT_FILE}"
        _drill_display "${EXAM_CURRENT_FILE}"
        ;;

      prev)   # EXAM-05
        exam_navigate "prev"
        exam_session_read
        scenario_load "${EXAM_CURRENT_FILE}"
        _drill_display "${EXAM_CURRENT_FILE}"
        ;;

      jump)   # EXAM-05
        if [[ -z "${2:-}" ]]; then
          error "Usage: ckad-drill exam jump <number>"
          exit "${EXIT_ERROR}"
        fi
        exam_navigate "$2"
        exam_session_read
        scenario_load "${EXAM_CURRENT_FILE}"
        _drill_display "${EXAM_CURRENT_FILE}"
        ;;

      flag)   # EXAM-06
        exam_flag
        info "Question flagged for review."
        ;;

      submit)  # EXAM-09, EXAM-10, EXAM-11, EXAM-12
        exam_session_read

        # Grade all questions
        local grade_result
        grade_result=$(exam_grade)

        local score passed domain_results
        score=$(printf '%s' "${grade_result}" | jq -r '.score')
        passed=$(printf '%s' "${grade_result}" | jq -r '.passed')
        domain_results=$(printf '%s' "${grade_result}" | jq -c '.domains')

        # Display results
        header "Exam Results"
        info ""
        printf '%s' "${grade_result}" | jq -r '.domains[] | "  Domain \(.domain): \(.passed)/\(.total) (\(.percent)%)"'
        info ""
        info "Overall Score: ${score}%"
        if [[ "${passed}" == "true" ]]; then
          success "RESULT: PASS"
        else
          error "RESULT: FAIL (66% required)"
        fi

        # Record to progress (EXAM-11)
        progress_record_exam "${score}" "${passed}" "${domain_results}"

        # Cleanup all namespaces (EXAM-12)
        exam_cleanup_all_namespaces
        info "All exam namespaces cleaned up."
        ;;

      *)
        error "Unknown exam subcommand: ${1:-}"
        info "Usage: ckad-drill exam [list|next|prev|jump N|flag|submit]"
        exit "${EXIT_ERROR}"
        ;;
    esac
    ;;
  ```
